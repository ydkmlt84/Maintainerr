// Generate a migration (run from repo root)
yarn --cwd apps/server migration:generate ./src/database/migrations/<migration_name>

// Show migration status (optional)
yarn --cwd apps/server migration:show

// Run pending migrations
yarn --cwd apps/server migration:run

// Revert the latest migration
yarn --cwd apps/server migration:revert

// Notes:
// - Migration scripts are wired through tools/typeOrmCli.ts.
// - Use ./src/database/migrations/... path for generation (relative to apps/server).
